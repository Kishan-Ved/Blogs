# Define a function to build Huffman codes and tables
def build_huffman_tree(frequency):
    heap = [[weight, [symbol, ""]] for symbol, weight in frequency.items()]
    heapq.heapify(heap)
    while len(heap) > 1:
        lo = heapq.heappop(heap)
        hi = heapq.heappop(heap)
        for pair in lo[1:]:
            pair[1] = '0' + pair[1]
        for pair in hi[1:]:
            pair[1] = '1' + pair[1]
        heapq.heappush(heap, [lo[0] + hi[0]] + lo[1:] + hi[1:])
    huffman_tree = sorted(heapq.heappop(heap)[1:], key=lambda p: (len(p[-1]), p))
    huffman_codes = {symbol: code for symbol, code in huffman_tree}
    huffman_table = {code: symbol for symbol, code in huffman_tree}
    return huffman_codes, huffman_table

def build_huffman_tree(frequency):

    heapq.heapify(heap)
    while len(heap) > 1:
        lo = heapq.heappop(heap)
        hi = heapq.heappop(heap)
        for pair in lo[1:]:
            pair[1] = '0' + pair[1]
        for pair in hi[1:]:

    huffman_codes = {symbol: code for symbol, code in huffman_tree}
    huffman_table = {code: symbol for symbol, code in huffman_tree}
    return huffman_codes, huffman_table

# Function to encode (r, s) pairs
def encode_rs_pairs(rs_pairs, huffman_codes_s, huffman_codes_r):
    encoded_data = []
    for r, s in rs_pairs:
        encoded_data.append(huffman_codes_r[r] + format(s, huffman_codes_s[s]))
    return "".join(encoded_data)

# Function to decode (r, s) pairs
def decode_rs_pairs(encoded_data, huffman_table_s, huffman_table_r):
    decoded_data = []
    bits_read = 0
    while bits_read < len(encoded_data):
        r = None
        for bit_length in range(1, 17):
            potential_r = encoded_data[bits_read:bits_read + bit_length]
            if potential_r in huffman_table_r:
                r = huffman_table_r[potential_r]
                bits_read += bit_length
                break
        s = ""
        while True:
            potential_s = encoded_data[bits_read:bits_read + 1]
            s += potential_s
            if s in huffman_table_s:
                s = huffman_table_s[s]
                bits_read += len(s)
                break
        decoded_data.append((r, int(s, 2)))
    return decoded_data
